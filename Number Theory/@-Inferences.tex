
\partl{Euler Function}

\partl{1.} function value of powers of prime numbers.
\begin{equation}
\varphi(p^k)=(p-1)\cdot p^{k-1}, p\in \{primes\}.
\end{equation}

\partl{2.} Euler theory in number theory.
\begin{equation}
\gcd(a,m)=1 \Rightarrow a^{\varphi(m)}\equiv 1\pmod m.
\end{equation}

\begin{equation}
\gcd(a,p)=1 \Rightarrow a^{b}\equiv a^{b \bmod \varphi(p)}\pmod p.
\end{equation}

\partl{3.} Decomposition induction.
\begin{equation}
n=\sum_{i=1}^s p_{i}^{k_i} \Rightarrow \varphi(n)=n\cdot \prod_{i=1}^s \frac{p_i -1}{p_i}.
\end{equation}

\partl{Fermat's little theorem}

\begin{equation}
\gcd(a,p)=1 \Rightarrow a^{p}\equiv a\pmod p.
\end{equation}
